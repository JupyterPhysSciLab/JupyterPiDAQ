275    def collectclick(self, btn):
276        if (btn.description == 'Start Collecting'):
277            btn.description = 'Stop Collecting'
278            btn.button_style = 'danger'
279            btn.tooltip = 'Stop the data collection'
280            # do not allow parameters to be reset after starting run.
281            self.setupbtn.disabled = True
282            self.setupbtn.tooltip = 'Parameters locked. The run has started.'
283            self.rateinp.disabled = True
284            self.timelbl.disabled = True
285            thread = threading.Thread(target=self.updatingplot, args=())
286            thread.start()
287            # self.updatingplot() hangs up user interface
288        else:
289            btn.description = 'Done'
290            btn.button_style = ''
291            btn.tooltip = ''
292            time.sleep(3)  # wait a few seconds for end of data collection
293            self.data = data
294            self.timestamp = timestamp
295            self.stdev = stdev
296            self.fillpandadf()
297            # save data to csv file so can be loaded elsewhere.
298            svname = self.title + '_' + time.strftime('%y-%m-%d_%H%M%S',
299                                                      time.localtime()) + '.csv'
300            self.pandadf.to_csv(svname)
301            self.collectbtn.close()
302            del self.collectbtn
303            #display(self.collecttxt)
304            display(HTML(
305                '<span style="color:blue;font-weight:bold;">DATA SAVED TO:' +
306                svname + '</span>'))
307            # Save the notebook with current widget states (plotly plots).
308            jscode = 'Jupyter.actions.call(' \
309                     '"widgets:save-with-widgets");'
310            display(JS(jscode))

312    def fillpandadf(self):
313        datacolumns = []
314        temptimes = np.transpose(self.timestamp)
315        tempdata = np.transpose(self.data)
316        tempstdev = np.transpose(self.stdev)
317        chncnt = 0
318        for i in range(self.ntraces):
319            if (self.traces[i].isactive):
320                chncnt += 1
321        for i in range(chncnt):
322            if self.ignore_skew and i > 0:
323                pass
324            else:
325                datacolumns.append(temptimes[i])
326            datacolumns.append(tempdata[i])
327            datacolumns.append(tempstdev[i])
328        titles = []
329        # Column labels.
330        for i in range(self.ntraces):
331            if (self.traces[i].isactive):
332                if self.ignore_skew:
333                    if i == 0:
334                        titles.append(self.timelbl.value)
335                else:
336                    titles.append(self.traces[
337                                  i].tracelbl.value + '_' + self.timelbl.value)
338                titles.append(
339                    self.traces[i].tracelbl.value + '(' + self.traces[
340                        i].units.value + ')')
341                titles.append(
342                    self.traces[i].tracelbl.value + '_' + 'stdev')
343        # print(str(titles))
344        # print(str(datacolumns))
345        self.pandadf = pd.DataFrame(np.transpose(datacolumns), columns=titles)

347    def updatingplot(self):
348        """
349        Runs until a check of self.collectbtn.desdcf63064cription does not return
350        'Stop Collecting'. This would probably be more efficient if set a
351        boolean.
352        """
353        starttime = time.time()
354        global data
355        data = []
356        global timestamp
357        timestamp = []
358        global stdev
359        stdev = []
360        datalegend = []
361        timelegend = []
362        stdevlegend = []
363        PLTconn, DAQconn = Pipe()
364        DAQCTL, PLTCTL = Pipe()
365        whichchn = []
366        gains = []
367        toplotx = []
368        toploty = []
369        nactive = 0
370        for k in self.traces:
371            if k.isactive:
372                nactive += 1
373        if self.separate_traces_checkbox.value:
374            self.livefig.set_subplots(rows = nactive, cols = 1,
375                                      shared_xaxes= True)
376            self.livefig.update_xaxes(title = self.timelbl.value,
377                                      row = nactive, col = 1)
378        else:
379            self.livefig.update_yaxes(title = "Values")
380            self.livefig.update_xaxes(title = self.timelbl.value)
381        active_count = 0
382        for i in range(self.ntraces):
383            if (self.traces[i].isactive):
384                whichchn.append({'board':self.traces[i].board,
385                                'chnl':self.traces[i].channel})
386                gains.append(self.traces[i].toselectedgain)
387                tempstr = self.traces[i].tracelbl.value + '(' + \
388                          self.traces[i].units.value + ')'
389                timelegend.append('time_' + tempstr)
390                datalegend.append(tempstr)
391                stdevlegend.append('stdev_' + tempstr)
392                if self.separate_traces_checkbox.value:
393                    scat = go.Scatter(y=[],x=[], name=tempstr)
394                    self.livefig.add_trace(scat, row = nactive-active_count,
395                                           col = 1)
396                    self.livefig.update_yaxes(title = self.traces[
397                        i].units.value, row = nactive-active_count,col = 1)
398                else:
399                    self.livefig.add_scatter(y=[],x=[], name=tempstr)
400                toplotx.append([])
401                toploty.append([])
402                active_count += 1
403        #print('whichchn: '+str(whichchn))
404        #print('gains: '+str(gains))
405        # Use up to 30% of the time for averaging if channels were spaced
406        # evenly between data collection times (with DACQ2 they appear
407        # more synchronous than that).
408        self.averaging_time = self.delta/nactive/3
409        DAQ = Process(target=DAQProc,
410                      args=(
411                      whichchn, gains, self.averaging_time, self.delta,
412                      DAQconn, DAQCTL))
413        DAQ.start()
414        lastupdatetime = time.time()
415
416        pts = 0
417        oldpts = 0
418        #print('about to enter while loop',end='')
419        while (self.collectbtn.description == 'Stop Collecting'):
420            #print('.',end='')
421            while PLTconn.poll():
422                pkg = PLTconn.recv()
423                self.lastpkgstr = str(pkg)
424                #print(self.lastpkgstr)
425                # convert voltage to requested units.
426                tmptime = 0
427                if self.ignore_skew:
428                    tmptime = sum(pkg[0]) / len(pkg[0])
429                for i in range(len(pkg[0])):
430                    avg = pkg[1][i]
431                    std = pkg[2][i]
432                    avg_std = pkg[3][i]
433                    avg_vdd = pkg[4][i]
434                    avg, std, avg_std = self.traces[
435                        self.tracemap[i]].toselectedunits(avg, std, avg_std, avg_vdd)
436                    avg, std, avg_std = sensors.to_reasonable_significant_figures_fast(
437                        avg, std, avg_std)
438                    pkg[1][i] = avg
439                    pkg[2][i] = std
440                    pkg[3][i] = avg_std
441                    if self.ignore_skew:
442                        toplotx[i].append(tmptime)
443                    else:
444                        toplotx[i].append(pkg[0][i])
445                    toploty[i].append(avg)
446                timestamp.append(pkg[0])
447                data.append(pkg[1])
448                stdev.append(pkg[3])
449            currenttime = time.time()
450            mindelay = 1.0
451            if self.separate_traces_checkbox.value:
452                mindelay = nactive*1.0
453            else:
454                mindelay = nactive*0.5
455            if (currenttime - lastupdatetime)>(mindelay+len(toplotx[0])*len(
456                    toplotx)/1000):
457                lastupdatetime = currenttime
458                for k in range(len(self.livefig.data)):
459                    self.livefig.data[k].x=toplotx[k]
460                    self.livefig.data[k].y=toploty[k]
461            #time.sleep(1)
462            PLTCTL.send('send')
463            time.sleep(self.delta)
464            # print ('btn.description='+str(btn.description))
465        endtime = time.time()
466        PLTCTL.send('stop')
467        time.sleep(0.5)  # wait 0.5 second to collect remaining data
468        PLTCTL.send('send')
469        time.sleep(0.5)
470        msg = ''
471        while (msg != 'done'):
472            while PLTconn.poll():
473                pkg = PLTconn.recv()
474                # print(str(pkg))
475                # convert voltage to requested units.
476                for i in range(len(pkg[0])):
477                    avg = pkg[1][i]
478                    std = pkg[2][i]
479                    avg_std = pkg[3][i]
480                    avg_vdd = pkg[4][i]
481                    avg, std, avg_std = self.traces[
482                        self.tracemap[i]].toselectedunits(avg, std, avg_std, avg_vdd)
483                    avg, std, avg_std = sensors.to_reasonable_significant_figures_fast(
484                        avg, std, avg_std)
485                    pkg[1][i] = avg
486                    pkg[2][i] = std
487                    pkg[3][i] = avg_std
488                    if self.ignore_skew:
489                        tmptime = sum(pkg[0])/len(pkg[0])
490                        toplotx[i].append(tmptime)
491                    else:
492                        toplotx[i].append(pkg[0][i])
493                    toploty[i].append(avg)
494                    #print(pkg[0][i])
495                    #print(avg)
496                timestamp.append(pkg[0])
497                data.append(pkg[1])
498                stdev.append(pkg[3])
499            PLTCTL.send('send')
500            time.sleep(0.2)
501            if PLTCTL.poll():
502                msg = PLTCTL.recv()
503                # print (str(msg))
504                if (msg != 'done'):
505                    print('Received unexpected message: ' + str(msg))
506        for k in range(len(self.livefig.data)):
507            self.livefig.data[k].x = toplotx[k]
508            self.livefig.data[k].y = toploty[k]
509        DAQ.join()  # make sure garbage collection occurs when it stops.
510        DAQconn.close()
511        PLTconn.close()
512        DAQCTL.close()
513        PLTCTL.close()
